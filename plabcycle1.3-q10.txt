d1={"a":10,"b":20,"c":30}
d2={"d":15,"e":25,"f":35}
d=d1.copy()
d.update(d2)
print("d1:",d1)
print("d2:",d2)
print("merged:",d)


Output:
d1: {'a': 10, 'c': 30, 'b': 20}                                                                                                         
d2: {'d': 15, 'f': 35, 'e': 25}                                                                                                         
merged: {'a': 10, 'f': 35, 'b': 20, 'c': 30, 'e': 25, 'd': 15}
